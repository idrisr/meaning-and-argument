\section{\S 7.1 Conditional Statements}
\begin{enumerate}

	\begin{statement}{If the book gets published, then Lucy will throw a party.}
		\item \symbol{A}{The book gets published}
		\item \symbol{B}{Lucy throws a party}
		\item \symbolization{A \implies B}
	\end{statement}

	\begin{statement}{If Rachel gets an A, Joan will be pleased.}
		\item \symbol{A}{Rachel gets an A}
		\item \symbol{B}{Joan is pleased}
		\item \symbolization{A \implies B}
	\end{statement}

	\begin{statement}{Mary will feed you if she likes you.}
		\item \symbol{A}{Mary feeds you}
		\item \symbol{B}{Mary likes you}
		\item \symbolization{A \implies B}
	\end{statement}

	\begin{statement}{Wearing clothes is necessary for going outside.}
		\item \symbol{A}{You go outside}
		\item \symbol{B}{You wear clothes }
		\item \symbolization{A \implies B}
	\end{statement}

	\begin{statement}{Bribing the judge is sufficient for going free.}
		\item \symbol{A}{You bribe the judge }
		\item \symbol{B}{You go free}
		\item \symbolization{A \implies B}
	\end{statement}

	\begin{statement}{John’s cooking dinner is necessary for Bill’s being happy.}
		\item \symbol{A}{Bill is happy}
		\item \symbol{B}{John cooks dinner}
		\item \symbolization{A \implies B}
	\end{statement}

\end{enumerate}
