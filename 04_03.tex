\section{\S 4.3 Complex Statements}
\begin{enumerate}

	\begin{statement}{Some of his students did not understand.}
		\item \symbol{A}{Some of his students did understand.}
		\item \symbolization{\neg A}
	\end{statement}

	\begin{statement}{That Jack ever slept here is unlikely.}
		\item \symbol{A}{It is likely Jack ever slept here.}
		\item \symbolization{\neg A}
	\end{statement}

	\begin{statement}{The solution must not be obvious.}
		\item \symbol{A}{The solution is obvious.}
		\item \symbolization{\neg A}
	\end{statement}

	\begin{statement}{John is unhappy about something.}
		\item \symbol{A}{John is unhappy about something.}
		\item \symbolization{\neg A}
	\end{statement}

	\begin{statement}{He saw nothing of interest in it.}
		\item \symbol{A}{He saw something of interest in it.}
		\item \symbolization{\neg A}
	\end{statement}

	\begin{statement}{John dislikes having to tell anyone what to do.}
		\item \symbol{A}{John dislikes having to tell anyone what to do.}
		\item \symbolization{A}
	\end{statement}

	\begin{statement}{I will force you to marry no one (ambiguous).}
		\item \symbol{A}{I will force you to marry someone}
		\item \symbol{B}{I will force you to marry someone}
		\item \symbolization{A}
	\end{statement}
