\begin{statement}{Nadia relaxes if and only if all her work is done.}
\symbol{A}{Nadia relaxes}
\symbol{B}{All Nadia's work is done}
\symbolization{A \implies B \land B \implies A}
\end{statement}

\begin{statement}{Since I went to the store, I will not run out of toothpaste.}
\symbol{A}{I went to the store}
\symbol{B}{I will run out of toothpaste}
\symbolization{A \implies \neg B}
\end{statement}

\begin{statement}{If you eat your dinner, then if thereâ€™s cake in the house, you can eat dessert.}
\symbol{A}{There's cake in the house}
\symbol{B}{you can eat your dinner}
\symbol{C}{you can eat dessert}
\symbolization{A \implies (B \implies C)}
\end{statement}

\begin{statement}{You can go out if, and only if, you clean your room.}
\symbol{A}{You can go out}
\symbol{B}{You clean your room}
\symbolization{A \iff B}
\end{statement}

\begin{statement}{Cleaning your room is necessary and sufficient for going out.}
\symbol{A}{Clean your room}
\symbol{B}{going out}
\symbolization{A \iff B}
\end{statement}
