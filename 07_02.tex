\section{\S 7.2 Only If, Unless}
\begin{enumerate}

\begin{statement}{Joe will wear his hat only if it rains.}
\symbol{A}{Joe wears his hat}
\symbol{B}{it rains}
\symbolization{A \implies B}
\end{statement}

\begin{statement}{Susan will be rich only if her aunt dies.}
\symbol{A}{Susan will be rich}
\symbol{B}{Susan's aunt dies}
\symbolization{A \implies B}
\end{statement}

\begin{statement}{I will live to be 100 unless I get hit by a sugar-crazed New
    York City taxi driver.}
\symbol{A}{I will live to be 100}
\symbol{B}{I get hit by a sugar-crazed New York City taxi driver}
\symbolization{\neg B \implies A}
\end{statement}

\begin{statement}{Laura will get an A only if she buys the teacher an apple.}
\symbol{A}{Laura gets an A}
\symbol{B}{Laura buys the teacher and apple}
\symbolization{A \implies B}
\end{statement}

\begin{statement}{Jerry will be sad unless his paper gets published.}
\symbol{A}{Jerry is sad}
\symbol{B}{Jerry's paper get published}
\symbolization{\neg B \implies A}
\end{statement}

\symbol{A}{I counted correctly}
\symbol{B}{There are three hundred cabbages in this garden}
\symbolization{A \implies B}
\end{statement}

\end{enumerate}
